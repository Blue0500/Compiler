func main() as void => {
    var x <- new OptionalInt { some = 4 };    
    # ref x <- 45;

    for i = 0 to 10 do {
        @i <- i - 1;
    };

    void;
};

union OptionalInt {
    none as void;
    some as int;
};

union Token {
    eof         as void;
    open_paren  as void;
    close_paren as void;
    plus        as void;
    minus       as void;
    times       as void;
    int_literal as int;
};
