function main () => int: {	
	var p <- new Point {
		x = 4
		y = 7
	}
	
	var c <- sum(p)

	mutate(@c)

	0
}

function mutate (var int some) => void: {
	@some <- 2 * some
}

function sum (Point p) => int: {
	p.x + p.y
}

function fib (int n) => int: {
	var a <- 0
	var b <- 1
	var count <- n

	while count do {
		var c <- a + b

		@a <- b
		@b <- c
		@count <- count - 1
	}

	count
}

struct Point {
	int x
	int y
}

struct A {
	var B b
}

struct B {
	var A a
}
